Compare Bremsstrahlung for several fuels.

l       = {'D-He3', 'D-T', 'p-B11'};
nI      = [1e20 1e20];
tE      = logspace(0,3);

pB = [];
pB(1,:) = Bremsstrahlung( nI, [1 2], tE, 2);
pB(2,:) = Bremsstrahlung( nI, [1 1], tE, 2);
pB(3,:) = Bremsstrahlung( nI, [1 5], tE, 2);

Plot2D(tE, pB, 'T_E (keV)','Power (MW/m^3)','Bremsstrahlung','log')
legend(l,'location','northwest');
