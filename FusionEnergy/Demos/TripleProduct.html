kB    = 1.3806503e-23; %m^2 kg/s^2 K
eVToK = 1.16045221e4 ;
beta  = 0.8;
b     = 0.12; % T
mu0   = 4e-7*pi;
T     = 1; % kEV
TK    = T*1000*eVToK;
n     = beta*b^2/(2*mu0)/(kB*TK);
tau   = 0.3;

nTTau = n*T*tau;

fprintf(1,'Triple Product %12.4e s kEv/m^3\n',nTTau)


%--------------------------------------

Triple Product   8.5827e+18 s kEv/m^3
